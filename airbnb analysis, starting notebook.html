# insure the host_since is in datetime
paris_listings['host_since']= pd.to_datetime(paris_listings['host_since'], errors='coerce')
# extract the year from 'host_since' column
paris_listings['host_since_year'] = paris_listings['host_since'].dt.year
# Group by the 'host_since_year' and calculqte the average price and count of rows
paris_listings_over_time=(
    paris_listings
    .groupby('host_since_year')
    .agg(
       average_price=('price', 'mean'),
        new_hosts=('host_since', 'count')
        
    )
    .reset_index()

)
